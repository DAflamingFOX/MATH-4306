\begin{align*}
    \text{Since } h(-x) & =-h(x)                                \\
    \text{And }|-x|     & =|x|                                  \\
    \text{Then }|h(x)|  & =|-h(x)|                              \\
    |-h(x)|             & = |h(-x)|                             \\
                        & \therefore\; |h(x)| \text{ is even. }
\end{align*}
