\begin{align*}
    \text{Since } h(-x)   & =-h(x)                               \\
    \text{And } g(x)      & =g(-x)                               \\
    \text{Then } g(h(-x)) & = g(-h(x))                           \\
    g(h(-x))              & = g(h(x))                            \\
                          & \therefore\; g(h(x)) \text{ is even}
\end{align*}
