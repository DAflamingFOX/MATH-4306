\begin{align*}
    f(x)      & = \int_{-\infty}^{\infty} C(\omega) \cdot e^{j \omega x}\,d\omega                                                           \\
    C(\omega) & = \frac{1}{2\pi} \int_{-\infty}^{\infty} f(x) \cdot e^{-j \omega x}\,dx                                                     \\
    C(\omega) & = \frac{1}{2\pi} \int_{0}^{\pi} \sin(x) \cdot e^{-j \omega x} \, dx                                                         \\
    C(\omega) & = \frac{1}{2\pi} \left[ \frac{e^{-j \omega x}}{1 - \omega^{2}} \left( -j \omega \sin(x) - \cos(x) \right) \right]_{0}^{\pi} \\
    C(\omega) & = \frac{1}{2\pi} \left( \frac{e^{-j \omega x}}{1 - \omega^{2}} - \frac{-1}{1 - \omega^{2}} \right)                          \\
    C(\omega) & = \frac{1}{2\pi} \left( \frac{e^{-j \omega x} + 1}{1 - \omega^{2}} \right)                                                  \\
    f(x)      & = \boxed{\frac{1}{2\pi} \int_{-\infty}^{\infty} \frac{e^{-j \omega x} + 1}{1 - \omega^{2}} \cdot e^{j \omega x} \, d\omega}
\end{align*}
