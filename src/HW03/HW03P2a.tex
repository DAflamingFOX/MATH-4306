    C(\omega) & = \frac{1}{2\pi} \int_{0}^{\infty} e^{-x} \cdot e^{-j \omega x} \, dx                            \\
    C(\omega) & = \frac{1}{2\pi} \int_{0}^{\infty} e^{-(1 + j \omega)x} \, dx                                    \\
    C(\omega) & = \frac{1}{2\pi} \left[ \frac{e^{-(1 + j \omega)x}}{-(1 + j \omega)} \right]_{0}^{\infty}        \\
    C(\omega) & = \frac{1}{2\pi} \left[ 0 - \frac{1}{-(1 + j \omega)} \right]                                    \\
    C(\omega) & = \frac{1}{2\pi (1 + j \omega)}                                                                  \\
    f(x)      & = \int_{-\infty}^{\infty} \frac{e^{j \omega x}}{2\pi(1 + j \omega)} \, d\omega                   \\
    f(x)      & = \boxed{\frac{1}{2\pi} \int_{-\infty}^{\infty} \frac{e^{j \omega x}}{1 + j \omega} \, d\omega }
