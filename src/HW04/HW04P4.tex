\begin{gather*}
    u(x,t)=v(x) + w(x,t)
\end{gather*}
\begin{align*}
    \diffp[2]{v}{x} & = -\frac{T}{a^{2}}                                                                                  \\
    v(x)            & = -\frac{T}{2a^{2}}x^{2} + C_{1}x + C_{2}                                                           \\
    v'(a)           & = 0 \Rightarrow 0 = -\frac{T}{a^{2}}a + C_{1} \Rightarrow C_{1} = \frac{T}{a}                       \\
    v(0)            & = T_{0} \Rightarrow T_{0} = -\frac{T}{2a^{2}}(0) + \frac{T}{a}(0) + C_{2} \Rightarrow C_{2} = T_{0} \\
    v(x)            & = -\frac{T}{2a^{2}}x^{2} + \frac{T}{a}x + T_{0}
\end{align*}
\begin{align*}
    w(x,t)          & = u(x,t) - v(x)                                                        \\
    \diffp[2]{w}{x} & = \frac{1}{k} \diffp{w}{t}                                             \\
    w(0,t)          & = u(0,t) - v(0) \Rightarrow T_{0} - T_{0} \Rightarrow w(0,t) = 0       \\
    w_{x}(a,t)      & = u_{x}(a,t) - v'(a) \Rightarrow 0 - 0 \Rightarrow w_{x}(a,t) = 0      \\
    w(x,0)          & = u(x,0) - v(x)                                                        \\
    w(x,0)          & = T_{0} - \left(  -\frac{T}{a^{2}}x^{2} + \frac{T}{a}x + T_{0} \right) \\
    w(x,0)          & = \frac{T}{2a^{2}}x^{2} - \frac{T}{a}x
\end{align*}
\begin{align*}
    \lambda_{n} & = \frac{(2n-1)\pi}{2a}            \\
    X_{n}(x)    & = \sin\left( \lambda_{n}x \right) \\
    Y_{n}(t)    & = e^{-\lambda_{n}^{2}kt}
\end{align*}
\begin{align*}
    w(x,t) & = \sum_{n=1}^{\infty} b_{n} \sin\left( \lambda_{n}x \right) e^{-\lambda_{n}^{2}kt}                                                                         \\
    w(x,0) & = \sum_{n=1}^{\infty} b_{n} \sin\left( \lambda_{n}x \right)                                                                                                \\
    b_{n}  & = \frac{2}{a}\int_{0}^{a} \left( \frac{T}{2a^{2}}x^{2} - \frac{T}{a}x \right) \sin\left( \lambda_{n}x \right)\,dx                                          \\
    I      & = \left[ \left( \frac{T}{2a^{2}}x^{2} - \frac{T}{a}x \right) \left( -\frac{1}{\lambda_{n}}\cos\left( \lambda_{n} x \right) \right)\right]_{0}^{a}
    - \int_{0}^{a} \left( -\frac{1}{\lambda_{n}}\cos\left( \lambda_{n} x \right) \right) \left( \frac{T}{a^{2}}x - \frac{T}{a} \right)\,dx                              \\
    I      & = \frac{1}{\lambda_{n}}\int_{0}^{a}  \left( \frac{T}{a^{2}}x - \frac{T}{a} \right) \cos\left( \lambda_{n} x \right)\,dx                                    \\
    I      & = \frac{1}{\lambda_{n}} \left( \left[ \left( \frac{T}{a^{2}}x - \frac{T}{a} \right) \frac{1}{\lambda_{n}} \sin\left( \lambda_{n}x \right)  \right]_{0}^{a}
    - \int_{0}^{a} \frac{1}{\lambda_{n}} \sin\left( \lambda_{n}x \right) \left( \frac{T}{a^{2}} \right)\,dx \right)                                                         \\
    I      & = \frac{1}{\lambda_{n}} \left( -\frac{T}{a^{2}\lambda_{n}} \int_{0}^{a}\sin\left( \lambda_{n}x \right)\,dx \right)                                         \\
    I      & = -\frac{T}{a^{2}\lambda_{n}^{2}} \int_{0}^{a}\sin\left( \lambda_{n}x \right)\,dx                                                                          \\
    I      & = -\frac{T}{a^{2}\lambda_{n}^{2}} \left[ -\frac{1}{\lambda_{n}} \cos \left( \lambda_{n}x \right) \right]_{0}^{a}                                           \\
    I      & = -\frac{T}{a^{2}\lambda_{n}^{2}} \frac{1}{\lambda_{n}}                                                                                                    \\
    I      & = -\frac{T}{a^{2}\lambda_{n}^{3}}                                                                                                                          \\
    b_{n}  & = \frac{2}{a} \cdot -\frac{T}{a^{2}\lambda_{n}^{3}}                                                                                                        \\
    b_{n}  & = -\frac{2T}{a^{3}\lambda_{n}^{3}}                                                                                                                         \\
    b_{n}  & = -\frac{2T}{a^{3}} \cdot \frac{1}{\left( \frac{(2n-1)\pi}{2a} \right)^{3}}                                                                                \\
    b_{n}  & = -\frac{2T}{a^{3}} \cdot \frac{8a^{3}}{(2n-1)^{3}\pi^{3}}                                                                                                 \\
    b_{n}  & = -\frac{16T}{(2n-1)^{3}\pi^{3}}
\end{align*}
\begin{gather*}
    u(x,t) = \boxed{-\frac{T}{2a^{2}}x^{2} + \frac{T}{a}x + T_{0} - \frac{16T}{\pi^{3}}\sum_{n=1}^{\infty} \frac{1}{(2n-1)^{3}} \sin\left( \frac{(2n-1)\pi x}{2a} \right) e^{-\left( \frac{(2n-1)\pi}{2a} \right)^{2}kt}}
\end{gather*}
